# assignments/task3/CMakeLists.txt

add_executable(task4 main.cpp 
    "${CMAKE_SOURCE_DIR}/src/demoShaderLoader.cpp"
    "${CMAKE_SOURCE_DIR}/src/openglDebug.cpp"
    "${CMAKE_SOURCE_DIR}/include/model/Point.h"
    "${CMAKE_SOURCE_DIR}/include/model/Ray.h"
    "${CMAKE_SOURCE_DIR}/include/model/LightSource.h"
    "${CMAKE_SOURCE_DIR}/include/model/Objects/Sphere.cpp"
    "${CMAKE_SOURCE_DIR}/include/operations/Shading.cpp"
    "${CMAKE_SOURCE_DIR}/include/model/Objects/Plane.cpp"
    "${CMAKE_SOURCE_DIR}/include/model/Objects/Cilinder.cpp"
    "${CMAKE_SOURCE_DIR}/include/model/Objects/Cone.cpp"
)

set_property(TARGET task4 PROPERTY CXX_STANDARD 17)

target_include_directories(task4 PUBLIC
    "${CMAKE_SOURCE_DIR}/include"
)

target_compile_definitions(task4 PUBLIC
    RESOURCES_PATH="${CMAKE_SOURCE_DIR}/resources/"
)

target_link_libraries(task4 PRIVATE
    glm glfw glad stb_image stb_truetype imgui Eigen3::Eigen
)


if(MSVC)
    target_compile_definitions(task4 PUBLIC _CRT_SECURE_NO_WARNINGS)
    set_property(TARGET task4 PROPERTY MSVC_RUNTIME_LIBRARY "MultiThreadedDebug<$<CONFIG:Debug>:Debug>")
    set_property(TARGET task4 PROPERTY MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Release>:Release>")
endif()
